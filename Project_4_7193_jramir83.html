System I: Recommendation Based on Popularity.##

Our get_popular_movies function determines popularity of movies based on the follwoing criteria:

  • Number of ratings,
  • Average rating above 3.5 stars,
  • and Minumum number of ratings to be considered.

We believe these criteria manage to build a list that is accurate and reflects an overall sense of what the public beleive is a popular movie. Popular movies tend to have a higher number of ratings as many it has reached a larger audience. Combined with the average rating above 3.5 stars, the list is narrowed to high audience films with really good feedback. The final criteria allows us to explore older films, as newer films usually have lower number of ratings.

def get_popular_movies(top_n=10):
    """
    Determine popular movies based on multiple criteria.

    Criteria for popularity:
    1. Number of ratings
    2. Average rating above 3.5 stars
    3. Minimum number of ratings to be considered
    """
    # Count number of non-NA ratings for each movie
    rating_counts = R.notna().sum()

    # Calculate average ratings for movies
    avg_ratings = R.mean()

    # Filter movies with at least 50 ratings and average rating > 3.5
    popular_movies = rating_counts[
        (rating_counts >= 50) &
        (avg_ratings >= 3.5)
    ]

    # Sort by rating count and average rating
    popular_movies_sorted = popular_movies.sort_values(ascending=False)

    # Get top N popular movies
    top_popular_movies = popular_movies_sorted.head(top_n)

    return top_popular_movies

import pandas as pd
import numpy as np
import multiprocessing as mp
from functools import partial

def compute_movie_similarities(R_normalized_data, movie_columns, i, num_movies):
    """
    Compute similarities for a specific movie against all other movies.

    Args:
    - R_normalized_data: Numpy array of normalized ratings
    - movie_columns: Column labels of the original matrix
    - i: Index of the current movie
    - num_movies: Total number of movies

    Returns:
    - Similarities for the current movie
    """
    similarities = np.full(num_movies, np.nan)

    for j in range(num_movies):
        if i == j:
            continue

        # Find common users (non-NaN entries) for both movies
        common_users = ~np.isnan(R_normalized_data[:, i]) & ~np.isnan(R_normalized_data[:, j])

        if np.sum(common_users) >= 3:
            # Extract ratings for the pair of movies
            Ri = R_normalized_data[common_users, i]
            Rj = R_normalized_data[common_users, j]

            # Compute cosine similarity
            numerator = np.sum(Ri * Rj)
            denominator = np.sqrt(np.sum(Ri ** 2)) * np.sqrt(np.sum(Rj ** 2))

            if denominator > 0:
                cosine_similarity = numerator / denominator
                transformed_similarity = 0.5 * (1 + cosine_similarity)
                similarities[j] = transformed_similarity

    return similarities

def compute_similarity_matrix(R_normalized, num_cores=None):
    """
    Compute movie-movie similarity matrix in parallel.

    Args:
    - R_normalized: Normalized rating matrix
    - num_cores: Number of CPU cores to use (defaults to all available)

    Returns:
    - DataFrame of movie-movie similarities
    """
    # Ensure multiprocessing safety
    mp.set_start_method('spawn', force=True)

    # Convert to numpy for faster processing
    R_normalized_data = R_normalized.to_numpy()
    movie_columns = R_normalized.columns

    # Use all available cores if not specified
    if num_cores is None:
        num_cores = mp.cpu_count()

    num_movies = R_normalized.shape[1]

    # Create a partial function with normalized data pre-filled
    compute_func = partial(
        compute_movie_similarities,
        R_normalized_data,
        movie_columns,
        num_movies=num_movies
    )

    # Use a Pool to parallelize computation
    with mp.Pool(processes=num_cores) as pool:
        # Compute similarities for each movie
        similarity_list = pool.map(compute_func, range(num_movies))

    # Convert to numpy array and create DataFrame
    S = np.array(similarity_list)

    # Ensure symmetry by taking the max of each pair
    S_symmetric = np.maximum(S, S.T)

    return pd.DataFrame(S_symmetric, index=movie_columns, columns=movie_columns)

def normalize_ratings(R):
    """
    Normalize the rating matrix by subtracting row means.
    Only considers non-NA entries when computing row means.
    """
    row_means = R.mean(axis=1, skipna=True)
    R_normalized = R.sub(row_means, axis=0)
    return R_normalized, row_means

def filter_top_k(S, k=30):
    """
    Retain the top k similarities for each movie.
    """
    S_top_k = S.apply(
        lambda row: row.nlargest(k).where(row.notna()), axis=1
    ).fillna(0)
    return S_top_k

def myIBCF(new_user, S, R):
    """
    Compute movie recommendations for a new user using Item-Based Collaborative Filtering.
    """
    predictions = {}
    for movie_id in S.columns:
        if pd.isna(new_user.get(movie_id, np.nan)):  # Predict only for unrated movies
            neighbors = S.loc[movie_id].dropna()  # Similar movies
            rated_neighbors = neighbors.index.intersection(new_user.dropna().index)
            if len(rated_neighbors) > 0:
                numerator = sum(neighbors[r] * new_user[r] for r in rated_neighbors)
                denominator = neighbors[rated_neighbors].sum()
                if denominator > 0:
                    predictions[movie_id] = numerator / denominator

    # Return top 10 predictions
    return pd.Series(predictions).sort_values(ascending=False).head(10)

    R.index = R.index.str.lstrip('u').astype(int)

    # Remove 'm' prefix from movie IDs in the column index
    R.columns = R.columns.str.lstrip('m').astype(int)

    # Preprocess ratings and compute similarity matrix
    R_normalized, row_means = normalize_ratings(R)
    S = compute_similarity_matrix(R_normalized, 12)

    S_df = pd.DataFrame(S_top30)

    S_df.to_csv('movie_similarity_matrix.csv', index=False)
